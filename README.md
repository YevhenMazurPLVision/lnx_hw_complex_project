# lnx_hw_complex_project

# To build everything:
make all


# To clean everything
make clean
make uninstall_convert


# To build task 1
make task1
# or
make calc


# To clean task 1
make clean
# or
make clean-task1
# or
make clean-calc


# To build task 2
make task2
make install_convert
# or
make upper
make lower
make install_conver

# Note that you can install only upper or lower
# Than you can clean them by one
make clean-upper
make uninstall_convert
# or
make clean-lower
make uninstall_convert

# To clean task 2
make clean
make uninstall_convert
# or
make clean-task2
make uninstall_convert
